# 2. 코드
### 1) python
```python
from collections import defaultdict

def solution(id_list, report, k):
    # 1. init
    res = []
    info = set()
    count = defaultdict(int)
    sent = defaultdict(list)
    mail = defaultdict(int)
    
    # 1. set count
    for r in report:
        if r in info: continue
            
        start, end = r.split(" ")
        count[end] += 1
        info.add(r)
        sent[end].append(start)
    
    # 2. count check
    for key, value in count.items():
        if value >= k:
            for sender in sent[key]:
                mail[sender] += 1

    # 3. mail count
    for id in id_list:
        res.append(mail[id])
    
    return res
```

### 2) java
```java
import java.util.*;

class Solution {
    public int[] solution(String[] id_list, String[] report, int k) {
        // 1. init
        int n = id_list.length;
        int[] res = new int[n];
        Map<String, Integer> count = new HashMap<>();
        Map<String, List<String>> sent = new HashMap<>();
        Map<String, Integer> mail = new HashMap<>();
        Set<String> s = new HashSet<>();

        // 2. set count
        for(String r: report) {
            if(s.contains(r)) continue;

            String[] info = r.split(" ");
            String start = info[0];
            String end = info[1];

            if(count.containsKey(end)) count.put(end, count.get(end) + 1);
            else count.put(end, 1);

            List<String> temp;
            if(sent.containsKey(end)) {
                temp = sent.get(end);
                temp.add(start);
                sent.put(end, temp);
            }
            else {
                temp = new ArrayList<>();
                temp.add(start);
                sent.put(end, temp);
            }

            s.add(r);
        }

        // 3. sender
        for(String id: id_list) {
            if(count.containsKey(id) && count.get(id) >= k) {
                for(String sender: sent.get(id)) {
                    if(mail.containsKey(sender)) mail.put(sender, mail.get(sender) + 1);
                    else mail.put(sender, 1);
                }
            }
        }

        // 4. res
        for(int i=0; i<n; i++) {
            String cur = id_list[i];
            if(mail.containsKey(cur)) res[i] = mail.get(cur);
        }

        return res;
    }
}
```
