# 2. 코드
### 1) Python
```python
def solution(n, t, m, p):

    def int_to_base_str(num, base):
        res = ""
        while num > 0:
            remain = num % base
            if remain >= 10:
                res += chr(ord('A') + remain % 10)
            else:
                res += str(remain)
            num //= base

        if res == "":
            res = "0"

        return res[::-1]

    r = ""
    for i in range(t * m):
        r += int_to_base_str(i, n)

    res = ""
    for i in range(p, t*m+1, m):
        idx = i - 1
        res += r[idx]

    return res
```

const solution = (n, t, m, p) => {

    const intToBaseStr = (num, base) => {
        let res = ""

        while(num > 0) {
            const remain = num % base
            if(remain >= 10) res += String.fromCharCode("A".charCodeAt(0) + (remain % 10))
            else res += String(remain)
            num = ~~(num / base)
        }

        if(res === "") res = "0"

        return res.split("").reverse().join("")
    }

    let r = ""
    for(let i=0; i<t*m; i++) {
        r += intToBaseStr(i, n)
    }

    let res = ""
    for(let i=p; i<=t*m; i += m) {
        res += r[i-1]
    }

    return res
}
