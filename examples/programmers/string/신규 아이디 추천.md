import re

def solution(new_id):
    answer = ''
    
    # 1. 
    new_id = new_id.lower()
    
    # 2. 
    new_id = re.sub('[^a-z0-9-_.]', '', new_id)

    # 3. 
    new_id = re.sub('[.]{2,}', '.', new_id)
    
    # 4. 
    new_id = re.sub('^[.]|[.]$', '', new_id)
    
    # 5. 
    new_id = 'a' if new_id == '' else new_id
    
    # 6. 
    new_id = new_id[:15]
    new_id = re.sub('[.]$', '', new_id)
    
    # 7. 
    while len(new_id) <= 2:
        new_id = new_id + new_id[-1:]
    
    return new_id
